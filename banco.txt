CREATE TABLE ALUNO (	
    NOME VARCHAR (255),
    EMAIL VARCHAR (255),
    DATA_NASCIMENTO DATE,
    ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY)


CREATE TABLE CURSOS(
    NOME VARCHAR (255),
    DURACAO VARCHAR (20),
    ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY
)


CREATE TABLE ALUNOS_CURSOS(
    ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    ID_ALUNO INT NOT NULL,
    ID_CURSO INT NOT NULL
)
